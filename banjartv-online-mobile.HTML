        async function fetchFromAPI(endpoint) {

            for (const proxy of PROXIES) {

                try {

                    const url = proxy ? proxy + encodeURIComponent(endpoint) : endpoint;

                    const controller = new AbortController();

                    const timeout = setTimeout(() => controller.abort(), 8000);

                    const res = await fetch(url, {

                        signal: controller.signal,

                        headers: { 'Accept': 'application/json' }

                    });

                    clearTimeout(timeout);

                    if (res.ok) {

                        const data = await res.json();

                        if (Array.isArray(data) && data.length > 0) return { data, proxy: proxy ? 'proxy' : 'direct' };

                    }

                } catch (e) {

                    // Lanjut ke proxy berikutnya

                }

            }

            return null;

        }

        function stripHtml(html) {

            if (!html) return '';

            const tmp = document.createElement('div');

            tmp.innerHTML = html;

            return tmp.textContent || tmp.innerText || '';

        }


        function formatDate(dateStr) {

            try {

                const d = new Date(dateStr);

                return d.toLocaleDateString('id-ID', { day: 'numeric', month: 'short', year: 'numeric' });

            } catch (e) {

                return dateStr;

            }

        }


        function timeAgo(dateStr) {

            try {

                const now = new Date();

                const d = new Date(dateStr);

                const diffMs = now - d;

                const diffMin = Math.floor(diffMs / 60000);

                const diffHr = Math.floor(diffMs / 3600000);

                const diffDay = Math.floor(diffMs / 86400000);

                if (diffMin < 1) return 'Baru saja';

                if (diffMin < 60) return diffMin + ' menit lalu';

                if (diffHr < 24) return diffHr + ' jam lalu';

                if (diffDay < 7) return diffDay + ' hari lalu';

                return formatDate(dateStr);

            } catch (e) {

                return dateStr;

            }

        }

        function parseWpPost(post) {

            const embedded = post._embedded || {};

            const media = embedded['wp:featuredmedia'] || [];

            const terms = embedded['wp:term'] || [];

            const categories = terms[0] || [];

            const author = embedded.author || [{}];


            const imageUrl = media[0] ? normalizeImageUrl(media[0].source_url || media[0].media_details?.sizes?.medium?.source_url) : null;

            const catName = categories.length > 0 ? categories[0].name : 'Umum';

            const catSlug = categories.length > 0 ? categories[0].slug : 'umum';


            return {

                id: post.id,

                title: stripHtml(post.title.rendered),

                excerpt: stripHtml(post.excerpt.rendered),

                content: post.content.rendered,

                date: post.date,

                dateFormatted: formatDate(post.date),

                timeAgo: timeAgo(post.date),

                image: imageUrl,

                category: catName,

                categorySlug: catSlug,

                author: author[0].name || 'Redaksi',

                link: post.link

            };

        }

        function startStream() {

            const video = document.getElementById('videoPlayer');

            document.getElementById('playOverlay').classList.add('hidden');

            document.getElementById('loadingSpinner').classList.add('show');

            document.getElementById('errorOverlay').classList.remove('show');

            updateStreamStatus('connecting', 'Menghubungkan...');


            if (Hls.isSupported()) {

                hls = new Hls({ enableWorker: true, lowLatencyMode: true, maxBufferLength: 30, startLevel: -1 });

                hls.loadSource(STREAM_URL);

                hls.attachMedia(video);

                hls.on(Hls.Events.MANIFEST_PARSED, () => {

                    video.play().catch(() => {});

                    onStreamStarted();

                });

                hls.on(Hls.Events.LEVEL_SWITCHED, (_, d) => {

                    const lv = hls.levels[d.level];

                    if (lv) {

                        let q = 'SD'; if (lv.height >= 1080) q = 'FHD'; else if (lv.height >= 720) q = 'HD'; else if (lv.height >= 480) q = 'SD'; else q = 'LD';

                        document.getElementById('qualityText').textContent = q + ' ' + lv.height + 'p';

                    }

                });

                hls.on(Hls.Events.ERROR, (_, d) => {

                    if (d.fatal) {

                        if (d.type === Hls.ErrorTypes.NETWORK_ERROR) { hls.startLoad(); setTimeout(() => { if (!isPlaying) showStreamError('Koneksi terputus. Periksa internet Anda.'); }, 8000); }

                        else if (d.type === Hls.ErrorTypes.MEDIA_ERROR) hls.recoverMediaError();

                        else showStreamError('Stream tidak tersedia.');

                    }

                });

            } else if (video.canPlayType('application/vnd.apple.mpegurl')) {

                video.src = STREAM_URL;

                video.addEventListener('loadedmetadata', () => { video.play().catch(() => {}); onStreamStarted(); });

                video.addEventListener('error', () => showStreamError('Format tidak didukung.'));

            } else {

                showStreamError('Browser tidak mendukung HLS.');

            }

        }
